At this point, one can modify the catalog.xml file to add/remove datasets that they want to distribute. This is IDD catalog.xml file that is included with the distribution: 


<?xml version="1.0" encoding="UTF-8"?>
<catalog xmlns="http://www.unidata.ucar.edu/namespaces/thredds/InvCatalog/v1.0"
  xmlns:xlink="http://www.w3.org/1999/xlink"
  name="Unidata THREDDS Data Server" version="1.0.1">

  <dataset name="Realtime data from IDD" ID="realtime_IDD">
    <metadata inherited="true">
      <authority>unidata.ucar.edu:</authority>
    </metadata>

    <catalogRef xlink:href="idd/models.xml" xlink:title="NCEP Model Data" />
    <catalogRef xlink:href="idd/radars.xml" xlink:title="NEXRAD Radar" />
    <catalogRef xlink:href="idd/obsData.xml" xlink:title="Station Data" />

    <catalogRef xlink:href="wcs/catalog.xml" xlink:title="WCS Data" />
  </dataset>


</catalog>
